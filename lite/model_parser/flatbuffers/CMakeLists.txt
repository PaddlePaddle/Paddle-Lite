function(lite_fbs_library TARGET)
  set(multiValueArgs SRCS FBS_DEPS)
  cmake_parse_arguments(args "" "" "${multiValueArgs}" ${ARGN})
  lite_cc_library(${TARGET} SRCS ${args_SRCS})
  add_dependencies(${TARGET} ${args_FBS_DEPS})
endfunction()
